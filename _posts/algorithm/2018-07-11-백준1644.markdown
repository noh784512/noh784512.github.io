문제 : <https://www.acmicpc.net/problem/1644>


풀이의 핵심 
+ 소수 구하기
	+   소수인지 아닌지 판별 -> 0~ 루트 n 까지 탐색
	+   주어진 수 까지의 모든 소수 구하기 ->[에라토스테네스의 체](https://ko.wikipedia.org/wiki/%EC%97%90%EB%9D%BC%ED%86%A0%EC%8A%A4%ED%85%8C%EB%84%A4%EC%8A%A4%EC%9D%98_%EC%B2%B4)
+ 부분합 구하기 


~~~
#include<iostream>
#include<cstring>
#include<math.h>
#include<vector>
#include<string>


using namespace std;

int N;
int arr[4000000];

void find_prime(int n)
{
	for (int i = 2; i*i<=n; i++)
	{
		if (arr[i])
		{
			for (int j = i * i; j <= n; j+=i)
				arr[j] = 0;
		}
	}
}

int main()
{
	int siz = 0;
	vector<int> v;
	cin >> N;
	memset(arr, 0, N+1 * sizeof(int));
	for (int i = 2; i <= N; i++)
	{
		arr[i] = 1;
	}
	find_prime(N);
	for (int i = 2; i <= N; i++)
	{
		if (arr[i])
		{
			v.push_back(i);
		}
	}

	int tot = 0;
	int sum = 0;
	int low = 0, high = 0;
	while(true)
	{
		if (sum>N)
		{
			sum = sum - v[low];
			low++;
		}
		else if(high == v.size())
		{
			break;
		}
		else 
		{
			sum = sum + v[high];
			high++;
		}
		if(sum==N)
		{
			tot++;
		}
	}
	cout << tot;
}
